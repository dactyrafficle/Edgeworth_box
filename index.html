the market for x

\dfrac{M}{p_x + p_y \cdot A ^ B} + \dfrac{M}{p_x + p_y \cdot A ^ B} = e_x + e_x

A = \frac{p_y}{p_x} \cdot \frac{\alpha}{\beta} , and B = \frac{1}{1-\delta}

multiply LHS by \frac{p_y}{p_y}

\dfrac{\frac{1}{p_y}}{\frac{1}{p_y}} \cdot \bigg ( \dfrac{p_x e_x + p_y e_y}{p_x + p_y \cdot A ^ B} + \dfrac{p_x e_x + p_y e_y}{p_x + p_y \cdot A ^ B} \bigg ) = e_x + e_x

let p = \frac{p_x}{p_y}

\dfrac{p \cdot e_x + e_y}{p + \Big ( \frac{\beta}{\alpha} \cdot p \Big )^{ \frac{1}{1-\delta} } } + \dfrac{p \cdot e_x + e_y}{p + \Big ( \frac{\beta}{\alpha} \cdot p \Big )^{ \frac{1}{1-\delta} } } = e_x + e_x

solve for p
